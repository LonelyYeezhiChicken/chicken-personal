рҹӨ– йҮҚж’ӯиҲҮеҒңжӯў whileиҝҙеңҲиҲҮbreak

ж—ҘеӯҗдёҖеӨ©дёҖеӨ©йҒҺпјҢ е‘ЁиҖҢеҫ©е§ӢпјҢз„ЎйҷҗеҫӘз’° C#иЈЎйқўд№ҹжңүйҮҚиӨҮдёҚж–·зҡ„иӘһеҸҘпјҢwhile

while иҝҙеңҲвҖӢ

  • еҹ·иЎҢеүҚжңғе…ҲеҲӨж–·жўқд»¶жҳҜеҗҰзӮәзңҹ
  • еҰӮжһңжўқд»¶зӮәзңҹе°ұз№јзәҢеҹ·иЎҢпјҢзӣҙеҲ°жўқд»¶и®ҠжҲҗеҗҰ
  • еҰӮжһңжўқд»¶ејҸзңҹпјҢзЁӢејҸжңғж°ёйҒ еҹ·иЎҢ

зҜ„дҫӢ:жҲ‘еҖ‘дҫҶеҖ’ж•ё 10~1пјҢж•ёеҲ° 1 жҷӮе°ұе–Ҡж–°е№ҙеҝ«жЁӮвҖӢ

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace CsharpDemo
{
    class Program
    {
        static void Main(string[] args)
        {
            //е…Ҳе®Је‘ҠдёҖеҖӢй–Ӣй—ң
            bool open = true;

            //е®Је‘ҠдёҖеҖӢеҖ’ж•ёз”Ёзҡ„ж•ҙж•ёи®Ҡж•ё
            int number = 10;

            //й–Ӣе§Ӣеҹ·иЎҢеҖ’ж•ё
            while (open)
            {
                //еҰӮжһңж•ёеҲ°1е°ұжҠҠй–Ӣй—ңй—ңжҺүпјҢдёӢж¬Ўе°ұдёҚжңғеҹ·иЎҢ
                if(number == 1)
                {
                    open = false;
                }

                //еҚ°еҮәз•¶еүҚж•ёеӯ—пјҢеҚ°е®ҢеҫҢе°ұжёӣдёҖ
                Console.WriteLine(number--);
            }
            //йғҪеҖ’ж•ёе®ҢеҫҢе°ұе–Ҡж–°е№ҙеҝ«жЁӮ
            Console.WriteLine("ж–°е№ҙеҝ«жЁӮ!");

            Console.ReadKey();
        }
    }
  }

breakвҖӢ

  • еҸӘиҰҒеңЁиҝҙеңҲдёӯеҠ е…Ҙ breakпјҢз„Ўи«–иҝҙеңҲжўқд»¶еҰӮдҪ•е°ұйғҪзӣҙжҺҘзөҗжқҹиҝҙеңҲ
  • еҚідҪҝжўқд»¶зӮә true йӮ„жҳҜжңғзөҗжқҹиҝҙеңҲ

зҜ„дҫӢ:жҲ‘еҖ‘дҫҶз”Ёеүӣеүӣзҡ„зҜ„дҫӢдҫҶеҜҰй©— break еҗ§вҖӢ

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace CsharpDemo
{
    class Program
    {
        static void Main(string[] args)
        {
            //е…Ҳе®Је‘ҠдёҖеҖӢй–Ӣй—ң
            bool open = true;

            //е®Је‘ҠдёҖеҖӢеҖ’ж•ёз”Ёзҡ„ж•ҙж•ёи®Ҡж•ё
            int number = 10;

            //й–Ӣе§Ӣеҹ·иЎҢеҖ’ж•ё
            while (open)
            {

                //еҚ°еҮәз•¶еүҚж•ёеӯ—
                Console.WriteLine(number);

                //еҰӮжһңж•ёеҲ°1е°ұbreakпјҢж”ҫеҫҢйқўзҡ„еҺҹеӣ жҳҜbreakеҫҢе°ұдёҚжңғеҹ·иЎҢдәҶ
                if (number == 1)
                {
                    //еҹ·иЎҢеҲ°йҖҷйӮҠе°ұдёӯж–·
                    break;
                }
                //breakеҫҢе°ұдёҚжңғеҹ·иЎҢдәҶпјҢжүҖд»ҘйҒһжёӣж”ҫеҲ°еҫҢйқў
                number--;
            }
            //йғҪеҖ’ж•ёе®ҢеҫҢе°ұе–Ҡж–°е№ҙеҝ«жЁӮ
            Console.WriteLine("ж–°е№ҙеҝ«жЁӮ!");

             //жҲ‘еҖ‘еҜ«зңӢзңӢй–Ӣй—ңзҡ„еҖј
             Console.WriteLine("й–Ӣй—ң = " + open);

            Console.ReadKey();
        }
    }
  }

  • д»–и·ҹ break дёҖжЁЈжҳҜдёӯж–·иҝҙеңҲ
  • дҪҶдёҚдёҖжЁЈзҡ„й»һпјҢд»–жҳҜдёӯж–·жң¬ж¬ЎиҝҙеңҲ
  • йҖҷж¬ЎиҝҙеңҲеҹ·иЎҢдёҚеҹ·иЎҢпјҢдҪҶдёӢж¬ЎйӮ„жңғеҹ·иЎҢ

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace CsharpDemo
{
    class Program
    {
        static void Main(string[] args)
        {
            //е®Је‘ҠдёҖеҖӢиЁҲж•ёз”Ёзҡ„ж•ҙж•ёи®Ҡж•ё
            int number = 0;

            //й–Ӣе§Ӣеҹ·иЎҢиЁҲж•ё
            while (number <11)
            {
                //е…ҲеҒҡеҠ дёҖ
                number++;
                //еҰӮжһңдёҚжҳҜеҒ¶ж•ёпјҢжң¬ж¬ЎеҫҢйқўе°ұдёҚеҹ·иЎҢдәҶ
                if (number % 2 !=0)
                {
                    continue;
                }
                //еҚ°еҮәз•¶еүҚж•ёеӯ—пјҢдёҰеҠ дёҖ
                Console.WriteLine(number);
            }

            Console.ReadKey();
        }
    }
  }
